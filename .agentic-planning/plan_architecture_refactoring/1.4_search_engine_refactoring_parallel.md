```typescript
/**

export class TextTokenizer {
  /**
   * –¢–æ–∫–µ–Ω–∏–∑–∞—Ü–∏—è —Ç–µ–∫—Å—Ç–∞ (lowercase, —Ä–∞–∑–¥–µ–ª–µ–Ω–∏–µ –ø–æ —Å–ª–æ–≤–∞–º)
   */
  static tokenize(text: string): string[] {
    return text
      .toLowerCase()
      .replace(/[_-]/g, ' ')
      .split(/\W+/)
      .filter((token) => token.length > 0);
  }

  /**
   * –ü–æ–ª—É—á–∏—Ç—å –∫—Ä–∞—Ç–∫–æ–µ –æ–ø–∏—Å–∞–Ω–∏–µ (–ø–µ—Ä–≤–æ–µ –ø—Ä–µ–¥–ª–æ–∂–µ–Ω–∏–µ)
   */
  static getShortDescription(description: string): string {
    const firstSentence = description.split(/[.!?]/)[0];
    return firstSentence ? firstSentence.trim() : description;
  }
}
```

```typescript
/**
 * –ü–æ—Å—Ç—Ä–æ–∏—Ç–µ–ª—å –∏–Ω–¥–µ–∫—Å–∞ –∏–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç–æ–≤

export class ToolIndexBuilder {
  constructor(private readonly toolRegistry: ToolRegistry) {}

  /**
   * –ü–æ—Å—Ç—Ä–æ–∏—Ç—å –∏–Ω–¥–µ–∫—Å –∏–∑ ToolRegistry
   */
  build(): StaticToolIndex[] {
    const index: StaticToolIndex[] = [];
    const tools = this.toolRegistry.getAllTools();

    for (const tool of tools) {
      const metadata = tool.getMetadata();
      const definition = metadata.definition;

      if (!definition.name || !definition.description || !metadata.category) {
        continue;
      }

      index.push({
        name: definition.name,
        category: metadata.category,
        tags: metadata.tags ? Array.from(metadata.tags) : [],
        isHelper: metadata.isHelper ?? false,
        nameTokens: TextTokenizer.tokenize(definition.name),
        descriptionTokens: TextTokenizer.tokenize(definition.description),
        descriptionShort: TextTokenizer.getShortDescription(definition.description),
      });
    }

    return index;
  }
}
```

#### 3. SearchCache (–Ω–æ–≤—ã–π)

```typescript
/**
 * LRU –∫–µ—à –¥–ª—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ –ø–æ–∏—Å–∫–∞
 *
 * –û—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ—Å—Ç—å (SRP):
 * - –¢–û–õ–¨–ö–û –∫–µ—à–∏—Ä–æ–≤–∞–Ω–∏–µ SearchResponse
 * - –¢–û–õ–¨–ö–û —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ —Ä–∞–∑–º–µ—Ä–æ–º –∫–µ—à–∞ (LRU)
 */
export class SearchCache {
  private cache: Map<string, SearchResponse>;
  private readonly maxSize: number;

  constructor(maxSize: number = 100) {
    this.cache = new Map();
    this.maxSize = maxSize;
  }

  /**
   * –ü–æ–ª—É—á–∏—Ç—å –∏–∑ –∫–µ—à–∞
   */
  get(key: string): SearchResponse | undefined {
    return this.cache.get(key);
  }

  /**
   * –°–æ—Ö—Ä–∞–Ω–∏—Ç—å –≤ –∫–µ—à (—Å LRU eviction)
   */
  set(key: string, value: SearchResponse): void {
    this.cache.set(key, value);

    // LRU eviction: —É–¥–∞–ª—è–µ–º –ø–µ—Ä–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç –µ—Å–ª–∏ –ø—Ä–µ–≤—ã—Å–∏–ª–∏ –ª–∏–º–∏—Ç
    if (this.cache.size > this.maxSize) {
      const firstKey = this.cache.keys().next().value;
      if (firstKey) {
        this.cache.delete(firstKey);
      }
    }
  }

  /**
   * –û—á–∏—Å—Ç–∏—Ç—å –∫–µ—à
   */
  clear(): void {
    this.cache.clear();
  }

  /**
   * –ü–æ–ª—É—á–∏—Ç—å —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É –∫–µ—à–∞
   */
  getStats(): { size: number; maxSize: number } {
    return {
      size: this.cache.size,
      maxSize: this.maxSize,
    };
  }

  /**
   * –ì–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞—Ç—å –∫–ª—é—á –∫–µ—à–∞ –∏–∑ –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ –ø–æ–∏—Å–∫–∞
   */
  static createKey(params: SearchParams): string {
    return JSON.stringify({
      q: params.query.toLowerCase().trim(),
      c: params.category,
      h: params.isHelper,
      l: params.limit,
      d: params.detailLevel || 'name_and_description',
    });
  }
}
```

#### 4. SearchResultFormatter (–Ω–æ–≤—ã–π)

```typescript
/**
 * –§–æ—Ä–º–∞—Ç–∏—Ä–æ–≤—â–∏–∫ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ –ø–æ–∏—Å–∫–∞
 *
 * –û—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ—Å—Ç—å (SRP):
 * - –¢–û–õ–¨–ö–û —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ SearchResult[] –≤ ToolSearchResultItem[]
 * - –¢–û–õ–¨–ö–û –ø—Ä–∏–º–µ–Ω–µ–Ω–∏–µ —É—Ä–æ–≤–Ω—è –¥–µ—Ç–∞–ª–∏–∑–∞—Ü–∏–∏
 * - –¢–û–õ–¨–ö–û lazy loading –ø–æ–ª–Ω—ã—Ö –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö
 */
export class SearchResultFormatter {
  constructor(
    private readonly staticIndex: StaticToolIndex[],
    private readonly toolRegistry: ToolRegistry
  ) {}

  /**
   * –§–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –ø–æ —É—Ä–æ–≤–Ω—é –¥–µ—Ç–∞–ª–∏–∑–∞—Ü–∏–∏
   */
  format(results: SearchResult[], detailLevel: DetailLevel): ToolSearchResultItem[] {
    return results.map((r) => this.formatResult(r, detailLevel));
  }

  private formatResult(result: SearchResult, detailLevel: DetailLevel): ToolSearchResultItem {
    const staticData = this.staticIndex.find((t) => t.name === result.toolName);

    if (!staticData) {
      return { name: result.toolName };
    }

    switch (detailLevel) {
      case 'name_only':
        return {
          name: staticData.name,
        };

      case 'name_and_description':
        return {
          name: staticData.name,
          description: staticData.descriptionShort,
          category: staticData.category,
          score: Math.round(result.score * 100) / 100,
        };

      case 'full':
        return this.formatFullDetails(result, staticData);
    }
  }

  private formatFullDetails(
    result: SearchResult,
    staticData: StaticToolIndex
  ): ToolSearchResultItem {
    // Lazy load –ø–æ–ª–Ω—ã—Ö –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö –∏–∑ ToolRegistry
    const tool = this.toolRegistry.getTool(staticData.name);
    const fullMetadata = tool?.getMetadata();

    return {
      name: staticData.name,
      description: fullMetadata?.definition.description || staticData.descriptionShort,
      category: staticData.category,
      tags: staticData.tags,
      inputSchema: fullMetadata?.definition.inputSchema,
      examples: fullMetadata?.examples,
      score: Math.round(result.score * 100) / 100,
      matchDetails: result.matchDetails,
    };
  }
}
```

#### 5. ToolSearchEngine (—É–ø—Ä–æ—â–µ–Ω–Ω—ã–π)

```typescript
/**
 * –ü–æ–∏—Å–∫–æ–≤—ã–π –¥–≤–∏–∂–æ–∫ –¥–ª—è tools (—É–ø—Ä–æ—â–µ–Ω–Ω—ã–π)
 *
 * –û—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ—Å—Ç—å (SRP):
 * - –¢–û–õ–¨–ö–û –∫–æ–æ—Ä–¥–∏–Ω–∞—Ü–∏—è –ø–æ–∏—Å–∫–∞ (–¥–µ–ª–µ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ)
 * - –¢–û–õ–¨–ö–û —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏—è –∏–Ω–¥–µ–∫—Å–∞ –ø–æ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º
 * - –¢–û–õ–¨–ö–û –ø—Ä–∏–º–µ–Ω–µ–Ω–∏–µ –ª–∏–º–∏—Ç–∞
 *
 * –ù–ï –æ—Ç–≤–µ—á–∞–µ—Ç –∑–∞:
 * - –ü–æ—Å—Ç—Ä–æ–µ–Ω–∏–µ –∏–Ω–¥–µ–∫—Å–∞ (–¥–µ–ª–µ–≥–∏—Ä—É–µ—Ç—Å—è ToolIndexBuilder)

 * - –§–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ (–¥–µ–ª–µ–≥–∏—Ä—É–µ—Ç—Å—è SearchResultFormatter)
 * - –¢–æ–∫–µ–Ω–∏–∑–∞—Ü–∏—é (–¥–µ–ª–µ–≥–∏—Ä—É–µ—Ç—Å—è TextTokenizer)
 */
export class ToolSearchEngine {
  private readonly staticIndex: StaticToolIndex[];
  private readonly cache: SearchCache;
  private readonly formatter: SearchResultFormatter;

  constructor(
    staticIndex: readonly StaticToolIndex[] | null,
    toolRegistry: ToolRegistry,
    private readonly searchStrategy: ISearchStrategy
  ) {
    // –ï—Å–ª–∏ —Å—Ç–∞—Ç–∏—á–µ—Å–∫–∏–π –∏–Ω–¥–µ–∫—Å –Ω–µ –ø—Ä–µ–¥–æ—Å—Ç–∞–≤–ª–µ–Ω, —Å—Ç—Ä–æ–∏–º –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–∏–π
    if (staticIndex) {
      this.staticIndex = Array.from(staticIndex);
    } else {
      const builder = new ToolIndexBuilder(toolRegistry);
      this.staticIndex = builder.build();
    }

    this.cache = new SearchCache(100);
    this.formatter = new SearchResultFormatter(this.staticIndex, toolRegistry);
  }

  /**
   * –í—ã–ø–æ–ª–Ω–∏—Ç—å –ø–æ–∏—Å–∫ —Å –∫–µ—à–∏—Ä–æ–≤–∞–Ω–∏–µ–º
   */
  search(params: SearchParams): SearchResponse {
    const cacheKey = SearchCache.createKey(params);

    // –ü—Ä–æ–≤–µ—Ä—è–µ–º –∫–µ—à
    const cached = this.cache.get(cacheKey);
    if (cached) {
      return cached;
    }

    // –í—ã–ø–æ–ª–Ω—è–µ–º –ø–æ–∏—Å–∫
    const results = this.performSearch(params);

    // –ö–µ—à–∏—Ä—É–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç
    this.cache.set(cacheKey, results);

    return results;
  }

  /**
   * –í—ã–ø–æ–ª–Ω–∏—Ç—å –ø–æ–∏—Å–∫ –±–µ–∑ –∫–µ—à–∏—Ä–æ–≤–∞–Ω–∏—è
   */
  private performSearch(params: SearchParams): SearchResponse {
    // –§–∏–ª—å—Ç—Ä—É–µ–º –∏–Ω–¥–µ–∫—Å –ø–æ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º
    const filtered = this.filterIndex(params);

    // –ï—Å–ª–∏ query –ø—É—Å—Ç–æ–π, –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –≤—Å–µ –æ—Ç—Ñ–∏–ª—å—Ç—Ä–æ–≤–∞–Ω–Ω—ã–µ
    let searchResults: SearchResult[];
    if (!params.query || params.query.trim() === '' || params.query === '*') {
      searchResults = filtered.map((tool) => ({
        toolName: tool.name,
        score: 1.0,
      }));
    } else {
      // –í—ã–ø–æ–ª–Ω—è–µ–º –ø–æ–∏—Å–∫ –ø–æ —Å—Ç—Ä–∞—Ç–µ–≥–∏—è–º
      searchResults = this.searchStrategy.search(params.query, filtered);
    }

    // –ü—Ä–∏–º–µ–Ω—è–µ–º –ª–∏–º–∏—Ç
    const totalFound = searchResults.length;
    const limited = searchResults.slice(0, params.limit || DEFAULT_TOOL_SEARCH_LIMIT);

    // –§–æ—Ä–º–∞—Ç–∏—Ä—É–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã
    const tools = this.formatter.format(
      limited,
      params.detailLevel || DEFAULT_TOOL_SEARCH_DETAIL_LEVEL
    );

    return { tools, totalFound };
  }

  /**
   * –§–∏–ª—å—Ç—Ä–æ–≤–∞—Ç—å –∏–Ω–¥–µ–∫—Å –ø–æ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º
   */
  private filterIndex(params: SearchParams): StaticToolIndex[] {
    let filtered = this.staticIndex;

    if (params.category) {
      filtered = filtered.filter((t) => t.category === params.category);
    }

    if (params.isHelper !== undefined) {
      filtered = filtered.filter((t) => t.isHelper === params.isHelper);
    }

    return filtered;
  }

  /**
   * –û—á–∏—Å—Ç–∏—Ç—å –∫–µ—à
   */
  clearCache(): void {
    this.cache.clear();
  }

  /**
   * –ü–æ–ª—É—á–∏—Ç—å —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É –∫–µ—à–∞
   */
  getCacheStats(): { size: number; maxSize: number } {
    return this.cache.getStats();
  }
}
```
