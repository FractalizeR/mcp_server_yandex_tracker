  static isAllSuccess<TKey, TValue>(results: BatchResult<TKey, TValue>): boolean {
    return results.every((r) => r.status === 'fulfilled');
  }

  /**
   * –ü—Ä–æ–≤–µ—Ä—è–µ—Ç, –µ—Å—Ç—å –ª–∏ —Ö–æ—Ç—è –±—ã –æ–¥–Ω–∞ –æ—à–∏–±–∫–∞

    return results.some((r) => r.status === 'rejected');
  }

  /**
   * –ü–æ–ª—É—á–∞–µ—Ç —Ç–æ–ª—å–∫–æ —É—Å–ø–µ—à–Ω—ã–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã

  static getSuccessfulResults<TKey, TValue>(results: BatchResult<TKey, TValue>): TValue[] {
    return results
      .filter(
        (r): r is Extract<BatchResult<TKey, TValue>[number], { status: 'fulfilled' }> =>
          r.status === 'fulfilled'

      .map((r) => r.value);
  }

  /**
   * –ü–æ–ª—É—á–∞–µ—Ç —Ç–æ–ª—å–∫–æ –æ—à–∏–±–∫–∏

    return results
      .filter(
        (r): r is Extract<BatchResult<TKey, TValue>[number], { status: 'rejected' }> =>
          r.status === 'rejected'

      .map((r) => r.reason);
  }

  /**
   * –ü–æ–ª—É—á–∞–µ—Ç —É—Å–ø–µ—à–Ω—ã–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã —Å –∏—Ö –∫–ª—é—á–∞–º–∏

    results: BatchResult<TKey, TValue>
  ): Array<{ key: TKey; value: TValue }> {
    return results
      .filter(
        (r): r is Extract<BatchResult<TKey, TValue>[number], { status: 'fulfilled' }> =>
          r.status === 'fulfilled'
      )
      .map((r) => ({ key: r.key, value: r.value }));

  /**
   * –ü–æ–ª—É—á–∞–µ—Ç –Ω–µ—É–¥–∞—á–Ω—ã–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã —Å –∏—Ö –∫–ª—é—á–∞–º–∏

      .filter(
        (r): r is Extract<BatchResult<TKey, TValue>[number], { status: 'rejected' }> =>
          r.status === 'rejected'

      .map((r) => ({ key: r.key, error: r.reason }));

   * –ì—Ä—É–ø–ø–∏—Ä—É–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –ø–æ —Å—Ç–∞—Ç—É—Å—É

    failed: Array<{ key: TKey; error: Error }>;
  } {
    return {
      successful: this.getSuccessfulEntries(results),
      failed: this.getFailedEntries(results),
    };
  }

  /**
   * –ü–æ–ª—É—á–∞–µ—Ç —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è
   */
  static getStats<TKey, TValue>(results: BatchResult<TKey, TValue>): {
    total: number;
    successful: number;
    failed: number;
    successRate: number;
  } {
    const total = results.length;
    const successful = results.filter((r) => r.status === 'fulfilled').length;
    const failed = results.filter((r) => r.status === 'rejected').length;

    return {
      total,
      successful,
      failed,
      successRate: total > 0 ? (successful / total) * 100 : 0,
    };
  }
}
